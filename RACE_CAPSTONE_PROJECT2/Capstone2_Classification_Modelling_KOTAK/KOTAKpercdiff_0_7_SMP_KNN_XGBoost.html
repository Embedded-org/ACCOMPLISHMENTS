  diff=Close_list[j+counter+9]-Close_list[j+counter+8]
  DIFF.append(diff)
  percdiff=(diff*100)/Close_list[j+counter+9]
  PERCDIFF.append(percdiff)
  if(percdiff>0.7):
    TARGET.append('pos')
  elif(percdiff<-0.7):
    TARGET.append('neg')
  else:
    TARGET.append('neu')
  j=j+1;
  if(j==total-(counter+9)):
    break
