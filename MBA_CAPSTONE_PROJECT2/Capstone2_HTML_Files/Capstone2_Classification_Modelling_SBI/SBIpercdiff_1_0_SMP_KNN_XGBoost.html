  DIFF.append(diff)
  percdiff=(diff*100)/Close_list[j+counter+9]
  PERCDIFF.append(percdiff)
  if(percdiff>1.0):
    TARGET.append('pos')
  elif(percdiff<-1.0):
    TARGET.append('neg')
  else:
    TARGET.append('neu')
  j=j+1;
  if(j==total-(counter+9)):
    break
